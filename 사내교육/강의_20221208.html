let promise = new Promise(function(resolve, reject) {
      setTimeout(() => resolve("완료!"), 1000);
    });
    
    // resolve 함수는 .then의 첫 번째 함수(인수)를 실행합니다.
    promise.then(
      result => alert(result), // 1초 후 "완료!"를 출력
      error => alert(error) // 실행되지 않음
    );

    console.log(1);
    let promise = new Promise(function(resolve, reject) {
        console.log(2);
        setTimeout(() => {
            console.log(4);
            resolve("완료!");
            console.log(5);
        }, 1000);
        console.log(3)
    });
    console.log(6);

    promise.then(
      (result) => {
          console.log(8);
          console.log(result);
      }, // 1초 후 "완료!"를 출력
      (error) => {
          console.log(9);
          console.log(error);
      }// 실행되지 않음
    );
    console.log(7);
